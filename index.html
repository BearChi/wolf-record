const { normalizeInput, compactSpaces, renderPromptText, splitKeywords, } = require('../../utils/parsers/textParser'); const slackApiClient = require('../notify/slackApiClient'); const { formatError, pruneMapByTtl } = require('../../utils/helpers'); const { logError, logExpected } = require('../../utils/errorLogger'); const { createExitPattern, removeSession, setSessionState } = require('../core/slackFlowCommon'); const SVN_UPDATE_PHASE = Object.freeze({ PENDING: 'pending', WAITING: 'waiting', DONE: 'done', }); // 產生版本資訊文字 const formatSvnVersionInfo = (projectName, svnInfo, allUpdateTypes) => { const lines = [`${projectName} 目前版本資訊：`]; for (const ut of allUpdateTypes) { const revision = svnInfo[ut.svnInfoKey]; if (ut.isProject) { lines.push(`・${ut.label}：r${revision}（${svnInfo.svnURL}）`); } else { lines.push(`・${ut.label}：r${revision}`); } } return lines.join('\n'); }; // 解析用戶輸入的更新項目名稱，回傳匹配到的 updateType 陣列 const parseUpdateSelection = (text, allUpdateTypes, gameName) => { const tokens = normalizeInput(text) .split(/[\/,，、|\s]+/) .map(token => normalizeInput(token)) .filter(Boolean); // 任一 token 為 "all" 時，直接回傳全部項目 if (tokens.some(t => t.toLowerCase() === 'all')) { return allUpdateTypes.map(ut => ({ ...ut, updateKey: ut.isProject ? gameName : ut.key, })); } const updateTypeMap = new Map(); for (const updateType of allUpdateTypes) { updateTypeMap.set(updateType.label.toLowerCase(), updateType); updateTypeMap.set(updateType.key.toLowerCase(), updateType); } const matched = []; const seen = new Set(); for (const token of tokens) { const lower = token.toLowerCase(); const found = updateTypeMap.get(lower); if (found && !seen.has(found.key)) { seen.add(found.key); matched.push({ ...found, // 專案的 updateType key 為 gameName updateKey: found.isProject ? gameName : found.key, }); } } return matched; }; // 將 step 結果寫入 session（回填下一步或建立 buildRequest） const applyStepResult = (session, step, value) => { const nextStep = session.pendingSteps[1]; // peek 下一步 switch (step.type) { case 'gameArg': { // 若下一步是同平台的 buildType / apkPath，回填 gameArg if (nextStep && nextStep.platform === step.platform && (nextStep.type === 'buildType' || nextStep.type === 'apkPath')) { nextStep.gameArg = value; } else { // 該平台最後一步（iOS 情境）：直接建立 buildRequest session.buildRequests.push({ platform: step.platform, gameArg: value, isBuildAAB: false, targetPath: '', }); } break; } case 'buildType': { session.buildRequests.push({ platform: step.platform, gameArg: step.gameArg, isBuildAAB: value === 'release', targetPath: null, // apkPath step 補上 }); break; } case 'apkPath': { // 找到最後加入的同平台 buildRequest，補上 targetPath const req = [...session.buildRequests].reverse() .find(r => r.platform === step.platform); if (req) req.targetPath = value; break; } case 'webPath': { // 找到預填的 Web buildRequest，補上 targetPath if (step.buildRequestIndex !== undefined) { session.buildRequests[step.buildRequestIndex].targetPath = value; } else { const req = [...session.buildRequests].reverse() .find(r => r.platform === 'web'); if (req) req.targetPath = value; } break; } } }; const isSvnUpdatePhase = (session, phase) => session.svnUpdatePhase === phase; const createMessageHandler = ({ targetChannel, debug, promptText, platformListText, unknownText, alreadyActiveText, unknownMaxCount, endText, triggerPattern, retryDedupTtlMs, parsePlatforms, cleanupExpiredSessions, activeBuildThreads, activeThreadFlows, getBotUserId, onBuildRequest, allowedProjects, buildSteps, getPrompt, parseReply, stopHandler, exitKeyword = 'exit', svnUpdatePromptText, svnUpdateSkipKeyword, allUpdateTypes, getSVNInfo, updateSVN, getAllBuildWorkerStatus, resolveProjectName, resourceBundleHandler, }) => { const exitPattern = createExitPattern(exitKeyword); const exitKeywordText = splitKeywords(exitKeyword)[0] || 'exit'; const debugLog = debug ? (msg, context = {}) => { const { projectName, threadTs, userId } = context; const prefix = projectName ? `${projectName} | thread=${threadTs ?? '-'} | user=${userId ?? '-'} | ` : ''; console.log(`[DEBUG_SLACK] ${prefix}${msg}`); } : () => {}; const processedEventIds = new Map(); const isDuplicateEvent = (eventId, now = Date.now()) => { if (!eventId) return false; pruneMapByTtl({ map: processedEventIds, ttlMs: retryDedupTtlMs, now, getTimestamp: (seenAt) => seenAt, }); if (processedEventIds.has(eventId)) return true; processedEventIds.set(eventId, now); return false; }; const hasBotMention = (text) => { const botUserId = getBotUserId(); return Boolean(botUserId) && text.includes(`<@${botUserId}>`); }; const extractTriggerProject = (text) => { const candidates = [normalizeInput(text), compactSpaces(text)]; for (const candidate of candidates) { const match = triggerPattern.exec(candidate); if (!match) continue; if (match.groups?.project) return match.groups.project; if (match[1]) return match[1]; } return null; }; const endSession = ({ threadTs, projectName, userId, reason }) => { if (reason) { debugLog(reason, { projectName, threadTs, userId }); } removeSession({ sessions: activeBuildThreads, threadTs, activeThreadFlows, flowType: 'build', }); }; const setSvnUpdatePhase = (session, nextPhase, context = {}) => setSessionState({ session, key: 'svnUpdatePhase', nextState: nextPhase, debugLog, context, label: 'svnUpdatePhase', }); const askProjectSelection = async ({ client, channel, threadTs }) => { const projectList = (allowedProjects || []).join('、'); await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: `請先選擇要打包的專案：${projectList}`, }); }; const sendVersionAndAskSvnUpdate = async ({ client, channel, threadTs, session, userId }) => { let versionText; try { const svnInfo = getSVNInfo(session.projectName); versionText = formatSvnVersionInfo(session.projectName, svnInfo, allUpdateTypes); debugLog('版本查詢完成', { projectName: session.projectName, threadTs, userId }); } catch (err) { debugLog( `版本查詢失敗：${formatError(err)}`, { projectName: session.projectName, threadTs, userId } ); // 查詢失敗時跳過更新流程，直接進入平台選擇 setSvnUpdatePhase(session, SVN_UPDATE_PHASE.DONE, { threadTs, userId, projectName: session.projectName }); await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: `版本查詢失敗：${formatError(err)}\n\n${renderPromptText(promptText, session.projectName, platformListText)}`, }); return; } const optionLabels = allUpdateTypes.map(t => t.label).join('、') + '、All(全部更新)'; await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: `${versionText}\n\n${svnUpdatePromptText}\n可選項目：${optionLabels}`, }); }; const replyUnknownInput = async ({ client, session, threadTs, channel, stage, endReason, userId, }) => { session.unknownCount += 1; try { await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: unknownText }); if (!debug && session.unknownCount >= unknownMaxCount) { await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: endText }); endSession({ threadTs, projectName: session.projectName, userId, reason: endReason, }); } } catch (err) { logError(`slackMessageHandler unknown input in ${stage}`, err, { threadTs, channel, projectName: session.projectName, }); } }; const submitBuildAndCloseSession = async ({ client, session, threadTs, channel, userId, buildRequests, skipped = [], errorStage, }) => { try { debugLog(`submitAll：${buildRequests.map(r => r.platform).join(', ')}`, { projectName: session.projectName, threadTs, userId, }); const result = await onBuildRequest({ projectName: session.projectName, buildRequests, }); session.serialIds = result.serialIds || []; let replyText = result.replyText; if (skipped.length > 0) { replyText += `\n已跳過：${skipped.join('、')}`; } replyText += '\n\n此對話串監聽已結束，需要打包請開新留言'; await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: replyText }); } catch (err) { logError(`slackMessageHandler ${errorStage}`, err, { threadTs, channel, projectName: session.projectName, }); await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: `打包請求處理失敗：${formatError(err)}`, }); } endSession({ threadTs, projectName: session.projectName, userId, reason: 'session 結束（打包完成）', }); }; return async ({ event, client, body }) => { const { subtype, bot_id: botId, channel, user, text, thread_ts: threadTs, ts, } = event; const eventId = body?.event_id; debugLog(`收到訊息 event_id=${eventId} user=${user} channel=${channel} ts=${ts} threadTs=${threadTs || '-'} subtype=${subtype || '-'} botId=${botId || '-'} text=${JSON.stringify(text)}`); if (subtype || botId) { debugLog(`略過：subtype=${subtype} botId=${botId}`); return; } if (isDuplicateEvent(eventId)) { debugLog(`略過：重複事件 event_id=${eventId}`); return; } cleanupExpiredSessions(); const normalizedText = normalizeInput(text || ''); const isTargetChannel = !targetChannel || channel === targetChannel; if (!isTargetChannel) { debugLog(`略過：非目標頻道 channel=${channel} targetChannel=${targetChannel}`); return; } const botUserId = getBotUserId(); const mentionFound = hasBotMention(normalizedText); debugLog(`Bot mention 識別：botUserId=${botUserId || '(尚未取得)'} 訊息中有 @Bot=${mentionFound} 正規化後文字=${JSON.stringify(normalizedText)}`); // ── 終止指令偵測（頻道層級：@Bot 終止 #xxx-nnn）── if (mentionFound && stopHandler.hasStopKeyword(normalizedText)) { const serialId = stopHandler.extractSerialId(normalizedText); debugLog(`偵測到終止指令：流水號=${serialId || '(未找到)'}`); if (serialId) { await stopHandler.handleStop({ client, channel, serialId }); return; } } // ── List 指令偵測（頻道層級：@Bot list）── const listMatch = /\blist\b/i.test(normalizedText); debugLog(`list 指令偵測：有 @Bot=${mentionFound} 訊息含 "list"=${listMatch}`); if (mentionFound && listMatch) { const workers = getAllBuildWorkerStatus(); debugLog(`執行 list：打包任務數=${workers.length}`); let listText; if (workers.length === 0) { listText = '目前無進行中的打包任務'; } else { const lines = [`目前打包任務（${workers.length} 筆）：`]; for (const w of workers) { const serial = w.serialId ? `（${w.serialId}）` : ''; const statusLabel = w.isBusy ? w.status : '排隊中'; lines.push(`・${w.buildGameArgName} ${w.platform}${serial} — ${statusLabel}`); } listText = lines.join('\n'); } const rootTs = threadTs || ts; await slackApiClient.sendThreadMessage({ client, channel, threadTs: rootTs, text: listText }); return; } // ── 觸發偵測 ── const compactText = compactSpaces(normalizedText); const matchNormal = triggerPattern.test(normalizedText); const matchCompact = triggerPattern.test(compactText); const isTrigger = mentionFound && (matchNormal || matchCompact); debugLog(`打包觸發偵測：有 @Bot=${mentionFound} 原文 match=${matchNormal} 去空格後 match=${matchCompact} 最終觸發=${isTrigger} triggerPattern=${triggerPattern}`); const rootTs = threadTs || ts; if (resourceBundleHandler) { const rbHandled = await resourceBundleHandler.tryHandle({ event, client, normalizedText, mentionFound, }); if (rbHandled) return; } if (isTrigger) { const activeFlow = activeThreadFlows.get(rootTs); if (activeFlow && activeFlow !== 'build') { await slackApiClient.sendThreadMessage({ client, channel, threadTs: rootTs, text: '此對話串已有進行中的流程，請先完成或等待結束。', }); return; } const existingSession = activeBuildThreads.get(rootTs); debugLog(`建立打包 session：rootTs=${rootTs} 已有 session=${!!existingSession} session 關閉中=${existingSession?.closing}`); if (existingSession && !existingSession.closing) { existingSession.lastActivityAt = Date.now(); await slackApiClient.sendThreadMessage({ client, channel, threadTs: rootTs, text: alreadyActiveText }); return; } const rawProject = extractTriggerProject(normalizedText) || ''; const triggerProject = resolveProjectName(rawProject); debugLog(`解析專案名稱：輸入="${rawProject}" 解析結果="${triggerProject}"`); debugLog('觸發打包', { projectName: triggerProject, threadTs: rootTs, userId: user }); const now = Date.now(); activeBuildThreads.set(rootTs, { channel, projectName: triggerProject, createdAt: now, lastActivityAt: now, unknownCount: 0, closing: false, pendingSteps: null, buildRequests: null, serialIds: [], svnUpdatePhase: SVN_UPDATE_PHASE.PENDING, }); activeThreadFlows.set(rootTs, 'build'); const session = activeBuildThreads.get(rootTs); if (!session.projectName) { await askProjectSelection({ client, channel, threadTs: rootTs }); return; } await sendVersionAndAskSvnUpdate({ client, channel, threadTs: rootTs, session, userId: user, }); return; } // ── @Bot 但未觸發任何指令：回覆可用指令提示 ── if (mentionFound && !threadTs) { const projectList = (allowedProjects || []).join('、'); const helpText = [ '可用指令：', `・打包<專案名> / 打包（如：打包Domino 或 打包）— 觸發打包流程`, `・資源包 [mp|texas]（同義：熱更包 / 熱更）— 觸發資源包流程`, `・終止 #流水號（如：終止 #android-001）— 終止進行中的打包`, `・（Thread 內）${exitKeywordText} — 快速結束目前對話監聽`, `・list — 列出目前進行中的打包任務`, '', `可用專案：${projectList}`, ].join('\n'); const rootTs = threadTs || ts; debugLog(`無匹配指令，回覆 help 提示`); await slackApiClient.sendThreadMessage({ client, channel, threadTs: rootTs, text: helpText }); return; } // ── Thread 回覆處理 ── if (!threadTs) { debugLog(`略過：一般頻道訊息，無 @Bot mention 也無對應 session`); return; } const session = activeBuildThreads.get(threadTs); debugLog(`Thread 回覆處理：threadTs=${threadTs} 找到進行中 session=${!!session} session 關閉中=${session?.closing}`); if (!session || session.closing) return; session.lastActivityAt = Date.now(); // ── Thread 內快速結束監聽（不終止打包） ── if (exitPattern.test(normalizedText)) { await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: endText }); endSession({ threadTs, projectName: session.projectName, userId: user, reason: '收到 exit 指令，結束 session', }); return; } // ── Thread 內終止偵測 ── if (stopHandler.hasStopKeyword(normalizedText)) { if (session.serialIds && session.serialIds.length > 0) { await stopHandler.handleStopAll({ client, channel, serialIds: session.serialIds }); } else { await slackApiClient.sendMessage({ client, channel, text: '已取消打包流程' }); } // 終止後無論 debug 與否都刪除 session endSession({ threadTs, projectName: session.projectName, userId: user, }); return; } // ── SVN 更新階段 ── if (isSvnUpdatePhase(session, SVN_UPDATE_PHASE.PENDING)) { if (!session.projectName) { const projectInput = extractTriggerProject(normalizedText) || normalizedText; const selectedProject = resolveProjectName(projectInput); if (!selectedProject) { await replyUnknownInput({ client, session, threadTs, channel, stage: 'project selection', endReason: 'session 結束（專案選擇階段連續不明輸入）', userId: user, }); return; } session.projectName = selectedProject; debugLog(`選擇專案：${selectedProject}`, { projectName: session.projectName, threadTs, userId: user, }); await sendVersionAndAskSvnUpdate({ client, channel, threadTs, session, userId: user, }); return; } const skipPattern = new RegExp(`^${svnUpdateSkipKeyword}$`, 'i'); if (skipPattern.test(normalizedText.trim())) { debugLog('跳過 SVN 更新', { projectName: session.projectName, threadTs, userId: user, }); setSvnUpdatePhase(session, SVN_UPDATE_PHASE.DONE, { threadTs, userId: user, projectName: session.projectName }); await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: renderPromptText(promptText, session.projectName, platformListText), }); return; } const selections = parseUpdateSelection(normalizedText, allUpdateTypes, session.projectName); if (selections.length === 0) { await replyUnknownInput({ client, session, threadTs, channel, stage: 'svn update phase', endReason: 'session 結束（SVN 更新階段連續不明輸入）', userId: user, }); return; } // 開始更新 setSvnUpdatePhase(session, SVN_UPDATE_PHASE.WAITING, { threadTs, userId: user, projectName: session.projectName }); const updateLabels = selections.map(s => s.label).join('、'); debugLog(`開始 SVN 更新：${updateLabels}`, { projectName: session.projectName, threadTs, userId: user, }); await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: `更新中：${updateLabels}...`, }); // 記錄更新前版號 let preUpdateInfo; try { preUpdateInfo = getSVNInfo(session.projectName); } catch (error) { logExpected('slackMessageHandler preUpdateInfo fallback', error, { threadTs, projectName: session.projectName, }); preUpdateInfo = {}; } // 並行執行所有更新 const results = await Promise.allSettled( selections.map(s => updateSVN(s.updateKey).then(rev => ({ ...s, newRevision: rev }))) ); // 組裝結果訊息（只列有更新的項目） const resultLines = []; const debugParts = []; for (let i = 0; i < results.length; i++) { const result = results[i]; const selection = selections[i]; const oldRev = preUpdateInfo[selection.svnInfoKey] || '?'; if (result.status === 'fulfilled') { const newRev = result.value.newRevision; resultLines.push(`・${selection.label}：r${oldRev} → r${newRev}`); debugParts.push(`${selection.label} r${oldRev}→r${newRev}`); } else { const errMsg = result.reason?.message || String(result.reason); resultLines.push(`・${selection.label}：更新失敗 — ${errMsg}`); debugParts.push(`${selection.label} 失敗`); } } const hasAnySuccess = results.some(r => r.status === 'fulfilled'); const header = hasAnySuccess ? '更新完成：' : '更新失敗：'; debugLog(`SVN 更新完成：${debugParts.join(', ')}`, { projectName: session.projectName, threadTs, userId: user, }); setSvnUpdatePhase(session, SVN_UPDATE_PHASE.DONE, { threadTs, userId: user, projectName: session.projectName }); await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: `${header}\n${resultLines.join('\n')}\n\n${renderPromptText(promptText, session.projectName, platformListText)}`, }); return; } // SVN 更新執行中，忽略非終止訊息 if (isSvnUpdatePhase(session, SVN_UPDATE_PHASE.WAITING)) { await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: '更新中，請稍候...', }); return; } // ── 情境 1：尚未選平台 ── if (!session.pendingSteps) { const platforms = parsePlatforms(normalizedText); return; } // ── 情境 2：消費 step 佇列 ── // 佇列已空（session 尚未清除，例如 debug 模式或時序問題）：忽略後續訊息 if (session.pendingSteps.length === 0) return; const currentStep = session.pendingSteps[0]; const parseResult = parseReply(currentStep, normalizedText); return; } // 還有下一步：送出 prompt await slackApiClient.sendThreadMessage({ client, channel, threadTs, text: getPrompt(session.pendingSteps[0]), }); }; }; module.exports = { createMessageHandler, };
